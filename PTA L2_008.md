    #ifndef C_L2_008_H
    #define C_L2_008_H

    #include <iostream>
    #include <algorithm>
    using namespace std;
    int main2(){
        string str;
        getline(cin,str);
        if (str.length() == 1){
            cout << 1;
            return 0;
        }
        int MaxLength = 0;
        for (int i = 1; i < str.length() - 1; ++i) {
            int length = 1;
            for (int j = 1; j < str.length(); ++j) {
                if (i - j >= 0 && i + j < str.length() && str[i - j] == str[i + j]) length += 2;
                else break;
            }
            MaxLength = max(MaxLength,length);
            length = 0;
            for (int k = 1; k < str.length(); ++k) {
                if (i - k >= 0 && i + k - 1 < str.length() && str[i - k] == str[i + k - 1]) length += 2;
                else break;
            }
            MaxLength = max(length,MaxLength);
        }
        cout << MaxLength;
        return 0;
    }

    #endif //C_L2_008_H
