    using namespace std;

    #include <stdio.h>
    #include <vector>
    #include <algorithm>

    struct node{
        int beg, end;
    };

    int n, m, a, b, k, np, v;
    int destroy[10005];
    vector<node> vec;

    int main(){
        scanf("%d %d", &n, &m);
        for (int i = 0; i < m; ++i) {
            scanf("%d %d", &a, &b);
            vec.push_back({a, b});
        }
        scanf("%d", &k);
        for (int i = 0; i < k; ++i) {
            scanf("%d", &np);
            fill(destroy, destroy + n + 1, 0);
            for (int j = 0; j < np; ++j) {
                scanf("%d", &v);
                destroy[v] = 1;
            }
            int isVic = 1;
            for (int j = 0; j < vec.size(); ++j) {
                if (!destroy[vec[j].beg] && !destroy[vec[j].end]){
                    isVic = 0;
                    break;
                }
            }
            if (i != 0) printf("\n");
            if (isVic) printf("YES");
            else printf("NO");
        }
        return 0;
    }
