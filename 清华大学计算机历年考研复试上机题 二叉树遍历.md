##### 用栈来模拟,遇到#就出栈

    #include <iostream>
    #include <string.h>
    #include <stack>

    using namespace std;

    string str;
    stack<char> stack1;

    int main(){
        str.resize(100);
        cin >> str;
        for (int i = 0; i < str.length(); ++i) {
            if (str[i] != '#') stack1.push(str[i]);
            else {
                if (!stack1.empty()){
                    cout << stack1.top() << " ";
                    stack1.pop();
                }
            }
        }
    }
